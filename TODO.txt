get grub files using WSL2
u will need to install gcc
make  a kernel.c
make a boot.s
make grub.cfg inside C:\Users\tavi2\Desktop\TaviOS\iso_root\boot\grub
make iso

nasm -f elf32 boot.s -o boot.o
x86_64-elf-gcc -m32 -ffreestanding -fno-pic -fno-builtin -fno-stack-protector -c kernel.c -o kernel.o
x86_64-elf-ld -m elf_i386 -T linker.ld -o kernel.elf boot.o kernel.o
grub-mkrescue -o tavios.iso iso_root
